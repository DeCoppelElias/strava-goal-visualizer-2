**Goal:** Add a Recharts pace chart to the dashboard: an area showing actual cumulative running distance overlaid on a dashed linear goal pace line.

**Architecture:** `PaceChart` is a self-contained component that takes `dailySeries` and `goalKm` as props. It builds a merged day-of-year dataset internally and reads CSS variable colors from `getComputedStyle` at render time (so it responds to theme changes). `DashboardPage` renders the chart between the stats row and goal edit card, inside its own card.

**Tech Stack:** React 18, TypeScript, Recharts 2.x, CSS custom properties

**Prerequisite:** Frontend Part 1 (`2026-06-06-task-5.4-frontend-stats-goal.md`) must be complete — `DashboardPage` must already fetch `PersonalDashboard` and expose `dashState.data.daily_series`.

### Task 2: Create `PaceChart` component

**Files:**
- Create: `frontend/src/components/PaceChart.tsx`

The chart uses a Recharts `ComposedChart`. The x-axis is day-of-year (1–365). `buildChartData` merges the actual `daily_series` points with 12 month-boundary pace points into one sorted array; each point carries both `actual` (km, undefined for months with no run) and `pace` (linear goal km at that day).

CSS variable values cannot be used as SVG `stroke`/`fill` string props — they are read via `getComputedStyle(document.documentElement)` at render time, which picks up whichever theme is currently active.

- [ ] **Step 1: Create `frontend/src/components/PaceChart.tsx`**

```tsx
import {
  Area,
  CartesianGrid,
  ComposedChart,
  Line,
  ResponsiveContainer,
  Tooltip,
  XAxis,
  YAxis,
} from 'recharts'
import type { DailyDistancePoint } from '../api/client'

interface Props {
  dailySeries: DailyDistancePoint[]
  goalKm: number
}

interface ChartPoint {
  day: number
  actual?: number
  pace: number
}

const MONTH_START_DAYS = [1, 32, 60, 91, 121, 152, 182, 213, 244, 274, 305, 335]
const MONTH_LABELS = ['Jan','Feb','Mar','Apr','May','Jun','Jul','Aug','Sep','Oct','Nov','Dec']

function toDayOfYear(dateStr: string, year: number): number {
  const jan1 = new Date(year, 0, 1).getTime()
  const d = new Date(dateStr + 'T00:00:00').getTime()
  return Math.floor((d - jan1) / 86400000) + 1
}

function buildChartData(dailySeries: DailyDistancePoint[], goalKm: number): ChartPoint[] {
  const year = new Date().getFullYear()
  const isLeap = (year % 4 === 0 && year % 100 !== 0) || year % 400 === 0
  const daysInYear = isLeap ? 366 : 365

  const paceAt = (day: number) =>
    Math.round((day / daysInYear) * goalKm * 10) / 10

  const byDay = new Map<number, ChartPoint>()

  // Seed month boundary points for the pace line
  for (const day of [...MONTH_START_DAYS, daysInYear]) {
    byDay.set(day, { day, pace: paceAt(day) })
  }

  // Overlay actual distance points
  for (const p of dailySeries) {
    const day = toDayOfYear(p.date, year)
    const existing = byDay.get(day)
    byDay.set(day, {
      day,
      actual: p.cumulative_km,
      pace: existing?.pace ?? paceAt(day),
    })
  }

  return [...byDay.values()].sort((a, b) => a.day - b.day)
}

function monthTickFormatter(day: number): string {
  const idx = MONTH_START_DAYS.indexOf(day)
  return idx >= 0 ? MONTH_LABELS[idx] : ''
}

export default function PaceChart({ dailySeries, goalKm }: Props) {
  const data = buildChartData(dailySeries, goalKm)
  const style = getComputedStyle(document.documentElement)
  const accent  = style.getPropertyValue('--accent').trim()  || '#4b8cf7'
  const border  = style.getPropertyValue('--border').trim()  || '#272c3d'
  const text3   = style.getPropertyValue('--text-3').trim()  || '#3d4358'
  const surface2 = style.getPropertyValue('--surface-2').trim() || '#1c2030'

  return (
    <ResponsiveContainer width="100%" height={240}>
      <ComposedChart data={data} margin={{ top: 8, right: 8, bottom: 0, left: 0 }}>
        <CartesianGrid strokeDasharray="3 3" stroke={border} vertical={false} />
        <XAxis
          dataKey="day"
          type="number"
          domain={[1, 365]}
          ticks={MONTH_START_DAYS}
          tickFormatter={monthTickFormatter}
          tick={{ fontSize: 11, fontFamily: 'JetBrains Mono, monospace', fill: text3 }}
          axisLine={false}
          tickLine={false}
        />
        <YAxis
          tick={{ fontSize: 11, fontFamily: 'JetBrains Mono, monospace', fill: text3 }}
          axisLine={false}
          tickLine={false}
          width={40}
          tickFormatter={(v: number) => String(v)}
        />
        <Tooltip
          contentStyle={{
            background: surface2,
            border: `1px solid ${border}`,
            borderRadius: 8,
            fontSize: 12,
            fontFamily: 'JetBrains Mono, monospace',
          }}
          labelFormatter={(day: number) => {
            const idx = MONTH_START_DAYS.findIndex(
              (d, i) => d <= day && (MONTH_START_DAYS[i + 1] ?? 366) > day,
            )
            return MONTH_LABELS[idx] ?? `Day ${day}`
          }}
          formatter={(value: number, name: string) => [
            `${value.toFixed(1)} km`,
            name === 'actual' ? 'Distance' : 'Goal pace',
          ]}
        />
        <Line
          dataKey="pace"
          stroke={text3}
          strokeDasharray="4 4"
          strokeWidth={1.5}
          dot={false}
          connectNulls
        />
        <Area
          dataKey="actual"
          stroke={accent}
          fill="rgba(75,140,247,0.08)"
          strokeWidth={2}
          dot={false}
          connectNulls
        />
      </ComposedChart>
    </ResponsiveContainer>
  )
}
```
